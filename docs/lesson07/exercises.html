The following examples may help you understand repetition and lists better.

Number Guessing Game

The program in example 1 is a number guessing game that repeatedly asks the user to guess a number between 1 and 100. At line 21 the program includes a while loop, which will cause the computer to repeatedly execute the statements at lines 22–35. The while loop will end when the user enters the correct number.



# Example 1

import random


def main():
    print("I'm thinking of a number between 1 and 100.")
    print("Try to guess it!")
    print()

    # Prepare a string that the input function will show to
    # the user during the first iteration of the while loop.
    prompt = "Please enter an integer between 1 and 100: "

    # Get a random integer between 1 and 100.
    answer = random.randint(1, 100)

    guess = -1
    attempts = 0

    while guess != answer:
        # Get a guess from the user.
        guess = int(input(prompt))
        attempts += 1
        guide = ""

        # Compare the user's guess to the answer.
        if guess < answer:
            guide = "too low"
        elif guess > answer:
            guide = "too high"

        # Prepare a string that the input function will show to
        # the user during the next iteration of the while loop.
        prompt = f"{guess} is {guide}. Please enter another integer: "

    print()
    print(f"{guess} is correct!")
    print(f"You used {attempts} guesses to guess the number.")


# Call main to start this program.
if __name__ == "__main__":
    main()

Prime Numbers

    
The following program asks a user for a positive integer and then
    prints all the prime integers between 2 and the number entered by
    the user.



# Example 2

import math

def main():
    print("This program generates a list of prime numbers.")
    n = int(input(
        "Enter a positive integer where the computer should stop: "))
    primes_list = calc_primes(n)
    print(primes_list)


def calc_primes(n):
    """Return a list of all prime numbers between 2 and n inclusive.
    This function works by assuming all positive integers are prime
    and then as the computer learns that a number is not prime, it
    marks that number by storing False for that number in a list.

    Parameter n: the largest integer to check if it is prime
    Return: a list of prime numbers
    """
    # Create an empty list that will store the prime numbers.
    primes_list = []

    # The range function returns a sequence of integers between
    # a start value and a stop value, but it excludes the stop
    # value. calc_primes is supposed to include the number in
    # its parameter n, so in the code below, we pass n_plus_1
    # to the range function instead of n.
    n_plus_1 = n + 1

    # Assume all non-negative integers are prime by creating a
    # list of n + 1 boolean variables that are all set to True.
    is_prime = [True] * n_plus_1

    # 0 and 1 are not prime, so mark them with False.
    is_prime[0] = False
    is_prime[1] = False

    # For each integer between 2 and sqrt(n), if the integer
    # is prime, append it to the list of prime numbers and
    # then mark all multiples of that integer as not prime.
    limit = math.isqrt(n) + 1
    for i in range(2, limit):
        if is_prime[i]:
            # i is prime, so append it to
            # the list of prime numbers.
            primes_list.append(i)

            # Mark all integers that are a
            # multiple of i as being not prime.
            for multiple in range(i + i, n_plus_1, i):
                is_prime[multiple] = False

    # Append the prime numbers that are between
    # sqrt(n) and n onto the list of primes.
    for i in range(limit, n_plus_1):
        if is_prime[i]:
            primes_list.append(i)

    # Return the list of prime numbers.
    return primes_list


# Call main to start this program.
if __name__ == "__main__":
    main()
