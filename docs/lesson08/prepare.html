try:
    # Normal code
    # This block must include code that falls into two groups:
    # 1. Code that may cause an exception to be raised
    # 2. code that depends on the results from the code in the first group
except ZeroDivisionError as err:
    # Code that is executed if the code in the try block caused a
    # ZeroDivisionError to be raised.
except OSError as err:
    # Code that is executed if the code in the try block caused an
    # OSError to be raised.
except ValueError as err:
    # Code that is executed if the code in the try block caused a
    # ValueError to be raised.
except (RuntimeError, TypeError, NameError) as err:
    # Code that is executed if the code in the try block caused a
    # RuntimeError, TypeError, or NameError to be raised.
except as err:
    # Code that is executed if the code in the try block caused any
    # excption to be raised that was not handled by one of the other
    # except blocks.
else:
    # Code that is executed after the code in the try block if the code
    # in the try block did not cause an exception to be raised.
finally:
    # Code that is executed after all the other code in try, except, and
    # else blocks regardless of whether an exception was raised or not.
